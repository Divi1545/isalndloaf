// Sidebar.js

import React from 'react';
import './islandloaf-theme.css';

const menuItems = [
  { label: "Overview" },
  { label: "Booking Manager" },
  { label: "Calendar Sync", active: true },
  { label: "Pricing Engine" },
  { label: "AI Marketing" },
  { label: "AI Features" },
  { label: "AI Agent Trainer" },
  { label: "Business Data" },
  { label: "Analytics & Reports" },
  { label: "Profile Settings" },
  { label: "Notifications & Logs" },
];

export default function Sidebar() {
  return (
    <aside className="sidebar">
      <div className="sidebar-header">IslandLoaf</div>
      <ul className="sidebar-menu">
        {menuItems.map((item, idx) => (
          <li
            key={idx}
            className={`sidebar-menu-item${item.active ? ' active' : ''}`}
          >
            {item.label}
          </li>
        ))}
      </ul>
    </aside>
  );
}
